import random
# ----------------------------------------//
# PROBLEM (Knapsack Problem)
# Decision Variables (Search Space)
# Obs: in the __main__ function
# Objective Function
# calculate fitness
# check if solution is admissible
def objective_function(solution, values):
fitness = 0
solution_size = len(solution)
for i in range(solution_size):
if solution[i]:
fitness += values[i]
return fitness
def is_admissible(solution, weights, max_limit):
weight_sum = 0
solution_size = len(solution)
for i in range(solution_size):
if solution[i]:
weight_sum += weights[i]
return weight_sum <= max_limit
# ----------------------------------------//
# SOLUTION
# Encoding => 1, 0, 0, 1, 0
def build_solution(encoding_size, weights, max_limit):
while True:
solution = []
possible_values = [True, False]
for i in range(encoding_size):
randomly_selected_value = random.choice(possible_values)
solution.append(randomly_selected_value)
if is_admissible(solution, weights, max_limit):
return solution # e.g. [True, False, True, True, False]
# ----------------------------------------//
# ALGORITHM (Hill Climbing)
# Neighborhood Function (Bit Flipping function)
def neighborhood_function(solution, values, weights, max_limit):
neighborhood = []
solution_size = len(solution)
for i in range(solution_size):
neighbor = solution.copy()
neighbor[i] = not neighbor[i]
if is_admissible(neighbor, weights, max_limit):
neighborhood.append(neighbor)
best_neighbor = get_best_neighbor(neighborhood, values)
return best_neighbor
def get_best_neighbor(neighborhood, values):
fitnesses = []
for neighbor in neighborhood:
neighbor_fitness = objective_function(neighbor, values)
fitnesses.append(neighbor_fitness)
best_neighbor_index = fitnesses.index(max(fitnesses))
return neighborhood[best_neighbor_index]
# Search Process
def search(encoding_size, values, weights, max_limit):
solution = build_solution(encoding_size, weights, max_limit)
current_fitness = objective_function(solution, values)
while True:
best_neighbor = neighborhood_function(solution, values, weights, max_limit)
best_neighbor_fitness = objective_function(best_neighbor, values)
print(f"Candidate Solution: fitness={objective_function(solution, values)}, \
\n{solution}")
print(f"Neighbor Solution: fitness={objective_function(best_neighbor, values)}, \
\n{best_neighbor}")
print("\n\n")
if best_neighbor_fitness >= current_fitness:
solution = best_neighbor.copy()
current_fitness = best_neighbor_fitness
else:
print(f"FINAL SOLUTION: fitness={objective_function(solution, values)}, \n {solution}")
return solution

# Decision Variables (Search Space)

values = [4, 2, 1, 10, 2] # values
